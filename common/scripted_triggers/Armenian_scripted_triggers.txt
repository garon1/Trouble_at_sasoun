ARM_can_centralise = {
	check_variable = { ARM_authority_level < 3 }
	ARM_cannot_centralise_state = no
	ARM_can_centralise_region = yes
}

ARM_can_centralise_region = {
	if = {
		limit = { ARM_is_autonomous_region = yes }
		ARM_can_centralise_low = yes
	}
	else = {
		ARM_can_centralise_medium = yes
	}
}

ARM_can_centralise_low = {
	if = {
		limit = { ARM_is_syrian_state = yes }
		check_variable = { ROOT.syr_integration_factor > 49 }
	}
	else_if = {
		limit = { ARM_is_greek_state = yes }
		check_variable = { ROOT.gre_integration_factor > 49 }
	}
	else_if = {
		limit = { ARM_is_kurdish_state = yes }
		check_variable = { ROOT.kur_integration_factor > 49 }
	}
	else_if = {
		limit = { ARM_is_arabian_state = yes }
		check_variable = { ROOT.hej_integration_factor > 49 }
	}
	else_if = {
		limit = { ARM_is_iraqi_state = yes }
		check_variable = { ROOT.irq_integration_factor > 49 }
	}
	else_if = {
		limit = { ARM_is_libyan_state = yes }
		check_variable = { ROOT.tri_integration_factor > 49 }
	}
	else_if = {
		limit = { ARM_is_egyptian_state = yes }
		check_variable = { ROOT.egy_integration_factor > 49 }
	}
}

	OR = {
